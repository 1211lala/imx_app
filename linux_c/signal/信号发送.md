## 接收函数
1. int kill(pid_t pid, int sig);
2. int raise(int sig);

## 实例
```C
#include "stdio.h"
#include <signal.h>
#include <unistd.h>
#include "sys/types.h"
#include <stdlib.h>
#if 0
/* ./bin/main  <进程号> <信号量值> */
int main(int argc, char **argv)
{
    if (argc < 3)
    {
        printf("input type error.   ./xxx   <pid>    <signum>\r\n");
        exit(-1);
    }
    int pid = atoi(argv[1]);
    int sig = atoi(argv[2]);
    int ret = kill(pid, 0);
    if (ret != 0)
    {
        perror(argv[1]);
        exit(-1);
    }
    printf("进程 %d 存在\r\n", pid);
    ret = kill(pid, sig);
    if (ret < 0)
    {
        perror(" ");
    }
    printf("进程 %d 已杀死\r\n", pid);
    return 0;
}

#else

void signal_callback(int signum)
{
    printf("signum: %d\r\n", signum);
}

int main(int argc, char **argv)
{
    signal(SIGINT, signal_callback);

    while (1)
    {
        sleep(1);

        raise(SIGINT);
    }

    return 0;
}
```
